    \begin{itemize}

        \item machine code details
        \end{itemize}
    \item want to detect/analyze malware
        \begin{itemize}
        \item won't have luxury of working with source code
        \item malware tries to avoid analysis

        \end{itemize}
    \end{itemize}
\end{frame}
