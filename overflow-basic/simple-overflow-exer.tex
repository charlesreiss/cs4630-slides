\begin{lrbox}{\GradeAssignmentCA}
\begin{lstlisting}[language=C++,style=smaller]
int GradeAssignment(FILE *in) {
  int scores[10]; char buffer[16];
  for (int i = 0; i < 10; ++i) {
    gets(buffer);
    scores[i] =
        GradeAnswer(buffer, i);
  }
  Process(scores);
}
\end{lstlisting}
\end{lrbox}
