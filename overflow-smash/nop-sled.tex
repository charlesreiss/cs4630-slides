
\begin{frame}[fragile,label=guessEasier]{making guessing easier (1)}
\lstset{language=myasm,style=smaller}
\begin{minipage}{0.5\textwidth}
normal shellcode: \\
\begin{lstlisting}
xor %eax, %eax
leaq command(%rip), %rbx
/* setup "exec" system call */
...
...
mov $11, %al
syscall

command: .ascii "/bin/sh"
\end{lstlisting}
\end{minipage}
\begin{minipage}{0.5\textwidth}
easier to `guess' shellcode: \\
\begin{lstlisting}
nop /* one-byte nop */
nop
nop
nop
nop
nop
nop
xor %eax, %eax
lea command(%rip), %rbx
...
...
command: .ascii "/bin/sh"
\end{lstlisting}
\end{minipage}
\end{frame}
